# Generate a histogram using awk

```bash
awk '{
  # Set group range
  x=$1 - $1 % 200
  arr[x] += 1
}
END {
  for (key in arr) {
    $2=sprintf("%-*s", arr[key], "");
    # Limit width
    if (arr[key] > 60) $2=substr($2, 0, 60) "..."
    gsub(" ", "=", $2);
    printf("%s\t%s\t%s\n", key, arr[key], $2)
  }
}' input_data.txt | sort -n
```
